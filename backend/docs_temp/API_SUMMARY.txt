## 🚨 CRITICAL BUSINESS RULES

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. EMPLOYER APPROVAL
   → New employers start with isApproved: false
   → Cannot post jobs until admin approves
   → Email notification sent on approval

2. APPLICATION UNIQUENESS
   → One application per job per applicant
   → Enforced by compound unique database index
   → Returns 409 Conflict error if duplicate

3. SHIFT CREATION
   → Only for accepted applications
   → Requires valid job, employer, and applicant
   → Date must be in the future

4. ATTENDANCE RULES
   → Check-in only allowed on shift date
   → Check-out requires prior check-in
   → Late if check-in >15 minutes after start time
   → Total hours auto-calculated on check-out

5. STATUS HISTORY TRACKING
   → All status changes recorded in history
   → Includes: status, updatedBy, timestamp, note
   → Immutable (append-only)

6. BLOCKING & DEACTIVATION
   → Blocked employers cannot post jobs or manage applications
   → Inactive applicants cannot apply or check-in
   → Existing shifts/applications remain
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

## 🔄 COMPLETE USER WORKFLOWS

### EMPLOYER WORKFLOW: Hiring Process
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. POST /api/auth/employer/signup
   → Account created (isApproved: false)
   → Email: "Pending approval"

2. Admin: PUT /api/admin/employers/:id/approve
   → isApproved: true
   → Email: "Account approved"

3. POST /api/jobs
   → Job posted (status: open)
   → Visible to all applicants

4. Applicants: POST /api/applications/:jobId
   → Applications received
   → Email: "New application"

5. PUT /api/applications/:id/accept
   → Application accepted
   → Email + SMS to applicant

6. POST /api/shifts
   → Shift created
   → Email + SMS to applicant

7. Track attendance and approve
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

### APPLICANT WORKFLOW: Job Application to Shift Completion
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. POST /api/auth/applicant/signup
   → Account created (active)
   → Email: "Welcome"

2. GET /api/jobs (with filters)
   → Browse available jobs

3. POST /api/applications/:jobId
   → Application submitted
   → Wait for employer decision

4. Employer accepts
   → Email + SMS notification

5. GET /api/shifts/applicant/my-shifts
   → View assigned shifts

6. PUT /api/shifts/:id/confirm
   → Confirm shift attendance

7. POST /api/attendance/:shiftId/checkin
   → Check-in with GPS location
   → Late detection automatic

8. POST /api/attendance/:shiftId/checkout
   → Check-out with GPS location
   → Total hours calculated

9. Employer approves attendance
   → Payment processed (if integrated)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
